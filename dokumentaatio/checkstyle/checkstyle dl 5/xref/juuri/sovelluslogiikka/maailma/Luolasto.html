148     /**
149      * Asettaa luolastoon annettujen koodinaattien kohdalle hirviön. Samalla
150      * luodaan myös taitelu-tapahtuma, joka liitetään hirviöön.
151      *
152      * Jos asetus ei onnistu, palautetaan false. Muuten palautetaan true.
153      *
154      * @param x annettu x-koordinaatti
155      * @param y annettu y-koordinaatti
156      * @return onnistuiko hirviön asetus
157      */
158     public boolean asetaHirvio(int x, int y) {
159         if (x < 0 || x >= leveys || y < 0 || y >= korkeus) {
160             return false;
161         }
162 
163         Hirvio hh = new Hirvio(null, "Noita");
164         Taistelu tt = new Taistelu(hh);
165         hh.setTapahtuma(tt);
166         hh.setSijainti(x, y);
167         koordinaatisto[x][y] = hh;
168 
169         return true;
170     }

274     /**
275      * Asettaa luolaston koordinaateihin (x, y) lukitun oven. Oven pystyy
276      * avaamaan avaimella, jonka nimi annetaan metodille parametrina.
277      *
278      * Metodi luo samalla ovenAvaus tapahtuman sekä liittää sen ja oven
279      * toisiinsa.
280      *
281      * Jos oven luominen ja asettaminen ei onnistu, palauttaa metodi false.
282      * Muuten palautetaan true.
283      *
284      * @param x annettu x-koordinaatti
285      * @param y annettu y-koordinaatti
286      * @param avaajanNimi lukitun oven avaajan nimi
287      * @return onnistuiko oven asetus
288      */
289     public boolean asetaLukittuOvi(int x, int y, String avaajanNimi) {
290         if (x < 0 || x >= leveys || y < 0 || y >= korkeus) {
291             return false;
292         }
293         YleisEsine avain = new YleisEsine(avaajanNimi, Esine.AVAIN);
294 
295         Ovi ovi = new Ovi(true, null, avain);
296         OvenAvaus ovenAvaus = new OvenAvaus(ovi);
297         ovi.setTapahtuma(ovenAvaus);
298         ovi.setSijainti(x, y);
299         koordinaatisto[x][y] = ovi;
300 
301         return true;
302     }

374     /**
375      * Piirtää luolaston käyttöliittymän piirtotaululle. Metodi kutsuu
376      * luolatossa olevien kohteiden piirtometodeja.
377      *
378      * @param g käyttöliittyymän grafiikkaolio
379      * @param mittaKaava mittakaava, jossa piirretään
380      */
381     public void piirra(Graphics g, int mittaKaava) {
382         for (int i = 0; i < leveys; i++) {
383             for (int j = 0; j < korkeus; j++) {
384                 int x = i * mittaKaava;
385                 int y = j * mittaKaava;
386                 koordinaatisto[i][j].piirra(g, mittaKaava, x, y);
387             }
388         }
