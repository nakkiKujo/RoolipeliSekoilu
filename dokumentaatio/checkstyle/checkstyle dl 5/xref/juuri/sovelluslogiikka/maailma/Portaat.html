4   import java.awt.Color;
5   import java.awt.Graphics;
6   
7   /**
8    * Portaat ovat luolaston kohde, jonka kautta pelaaja voi edetä syvemmälle 
9    * luolaan. Edetessään portaissa ei pelaaja voi enää palata takaisin ylemmälle
10   * luolaston tasolle.
11   */
12  public class Portaat extends Kohde {
13      
14      public Portaat() {
15          this.koodi = Kohde.PORTAAT;
16          this.nimi = "portaat";
17          this.voikoKulkea = true;
18      }
19  
20      @Override
21      public void piirra(Graphics g, int mittakaava, int x, int y) {
22          g.setColor(Color.ORANGE);
23          g.fillRect(x, y, mittakaava, mittakaava);
24      }
25              
26  }
