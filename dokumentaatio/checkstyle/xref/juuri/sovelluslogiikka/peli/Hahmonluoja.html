10  public class Hahmonluoja {
11  
12      private Lukija lukija;
13      private Keskustelija keskustelija;
14  
15      public Hahmonluoja(Lukija ll, Keskustelija kk) {
16          this.lukija = ll;
17          this.keskustelija = kk;
18      }
19  
20      public Hahmo luoHahmo() {
21          String nimi = hahmonNimenValinta();
22          String hahmonAmmatti = hahmonAmmatinValinta();
23  
24          return luodaanHahmoAmmatinMukaan(hahmonAmmatti, nimi);
25      }
26  
27      private String hahmonNimenValinta() {
28          System.out.println("~~~~~~~~~~~~~~");
29          System.out.print("Anna ensin hahmollesi nimi: ");
30          String nimi = lukija.lueString();
31          nimi.trim();
32          System.out.println("Ah, " + nimi + ", siinäpä oikein osuva nimi urhealle pelurille!");
33          return nimi;
34      }
35  
36      private String hahmonAmmatinValinta() {
37  
38          //tämä täytyy muokata niin, että pystytään laajentamaan helpommin
39          System.out.println(keskustelija.ammatinValintaEsittely());
40          String teksti = lukija.lueString();
41          teksti.toLowerCase();
42  
43          // palautetaan ammatti stringinä
44          return kasitteleAmmatinValintaTeksti(teksti);
45      }
46  
47      private String kasitteleAmmatinValintaTeksti(String teksti) {
48          while (!(teksti.equals("soturi")) && !(teksti.equals("velho")) && !(teksti.equals("druidi"))) {
49              if ((teksti.equals("info velho")) || teksti.equals("info soturi") || teksti.equals("info druidi")) {
50                  System.out.println(keskustelija.ammattiInfo(teksti));
51                  System.out.println("Anna uusi komento: ");
52              } else {
53                  System.out.print("Hups, annoit väärän komennon! Kokeile uudestaan: ");
54              }
55  
56              teksti = lukija.lueString();
57              teksti.toLowerCase();
58          }
59  
60          return teksti;
61      }
62  
63      private Hahmo luodaanHahmoAmmatinMukaan(String ammatti, String hahmonNimi) {
64  
65          //oikaistaan hieman ja luodaan suoraan vain soturi
66          return new Soturi(hahmonNimi);
67      }
