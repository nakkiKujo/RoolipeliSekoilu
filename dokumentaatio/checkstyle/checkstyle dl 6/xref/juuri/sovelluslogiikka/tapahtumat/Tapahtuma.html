1   package juuri.sovelluslogiikka.tapahtumat;
2   
3   import juuri.sovelluslogiikka.hahmo.Hahmo;
4   import juuri.sovelluslogiikka.maailma.Luolasto;
5   
6   /**
7    * Tapahtumat liittyvä pelaajan hahmon ja luolaston kohteiden välisiin
8    * interaktioihin. Tapahtuma ensin valmistellaan. Tämän jälkeen on kaksi
9    * vaihtoehtoa, miten tapahtuma voi edetä.
10   */
11  public abstract class Tapahtuma {
12  
13      public static final int TAISTELU = 1;
14      public static final int OVENAVAUS = 2;
15      public static final int PORTAISSAKULKEMINEN = 3;
16      public static final int AARTEENLOYTAMINEN = 4;
17      public static final int ANSAANASTUMINEN = 5;
18  
19      protected int koodi;
20      protected String tapahtumaTeksti;
21      protected String vaihtoehtoYksi;
22      protected String vaihtoehtoKaksi;
23  
24      /**
25       * Metodi palauttaa tekstin, joka tapahtumalle asetetaan sitä
26       * valmistellessa. Tämä teksti näytetään pelaajalle käyttöliittymän
27       * välityksellä.
28       *
29       * @return tapahtumaan liittyvä teksti
30       */
31      public String getTapahtumateksti() {
32          return tapahtumaTeksti;
33      }
34  
35      /**
36       * Metodi palauttaa tekstin, joka liittyy tapahtuman ensimmäiseen tapaan
37       * ratketa. Tämä teksti näytetään pelaajalle käyttöliittymän välityksellä.
38       *
39       * @return vaihtoehtoon liittyvä teksti
40       */
41      public String getVaihtoehtoYksi() {
42          return vaihtoehtoYksi;
43      }
44  
45      /**
46       * Metodi palauttaa tekstin, joka liittyy tapahtuman toiseen tapaan ratketa.
47       * Tämä teksti näytetään pelaajalle käyttöliittymän välityksellä.
48       *
49       * @return vaihtoehtoon liittyvä teksti
50       */
51      public String getVaihtoehtoKaksi() {
52          return vaihtoehtoKaksi;
53      }
54  
55      /**
56       * Metodi palauttaa tapahtumalle ominaisen koodin. Erilaisilla tapahtumilla
57       * on erilaiset koodit.
58       *
59       * @return tapahtuman koodi
60       */
61      public int getKoodi() {
62          return koodi;
63      }
64  
65      /**
66       * Metodi suoritetaan hahmon kohdatessa tapahtumaan liittyvän kohteen. Se
67       * valmistelee kaiken valmiiksi tapahtumaa varten, ja muun muassa muokkaa
68       * tapahtumaan liittyvät tekstit hahmon ja kohteen mukaisiksi.
69       *
70       * @param hahmo pelaajan hahmo
71       */
72      public abstract void valmisteleTapahtuma(Hahmo hahmo);
73  
74      /**
75       * Metodi suoritetaan, kun tapahtuma tavalla tai toisella päätyy tapahtumaan
76       * tavalla yksi.
77       *
78       * @param hahmo pelaajan hahmo
79       * @param luola luolasto, jossa pelaajan hahmo ja tapahtuman kohde ovat
80       */
81      public abstract void toteutaVaihtoehtoYksi(Hahmo hahmo, Luolasto luola);
82  
83      /**
84       * Metodi suoritetaan, kun tapahtuma päätyy tapahtumaan tavalla numero
85       * kaksi.
86       *
87       * @param hahmo pelaajan hahmo
88       * @param luola luolasto, jossa pelaajan hahmo ja tapahtumaan liittyvä kohde
89       * ovat
90       */
91      public abstract void toteutaVaihtoehtoKaksi(Hahmo hahmo, Luolasto luola);
92  }
