3   import java.awt.Color;
4   import java.awt.Graphics;
5   import java.util.ArrayList;
6   import juuri.sovelluslogiikka.esineet.Esine;
7   import juuri.sovelluslogiikka.hahmo.Hahmo;
8   import juuri.sovelluslogiikka.tapahtumat.Tapahtuma;

10  /**
11   * Aarteet ovat luolaston kohteita, joista pelaaja voi saada lisää esineitä.
12   */
13  public class Aarre extends Kohde {
14  
15      private ArrayList<Esine> aarteet;
16  
17      /**
18       * Luo uuden Aarteen. Parametrina annetaan aarteen nimi. Lisäksi metodi luo
19       * aarteella listan sen sisältämistä esineistä. Lista on aluksi tyhjä.
20       *
21       * @param nimi aarteen nimi
22       */
23      public Aarre(String nimi) {
24          this.nimi = nimi;
25          this.koodi = Kohde.AARRE;
26          this.voikoKulkea = false;
27          this.aarteet = new ArrayList<>();
28      }
29  
30      /**
31       * Metodi lisää esineen aarteeseen.
32       *
33       * @param ee lisättävä esine
34       */
35      public void lisaaAarre(Esine ee) {
36          this.aarteet.add(ee);
37      }
38  
39      /**
40       * Metodi palauttaa aarteessa olevat esineet ja samalla poistaa esineet
41       * aarteesta.
42       *
43       * @return lista aarteen esineistä
44       */
45      public ArrayList<Esine> keraaAarteet() {
46          ArrayList<Esine> apu = new ArrayList<>();
47  
48          for (Esine aarteenEsine : aarteet) {
49              apu.add(aarteenEsine);
50          }
51  
52          aarteet.clear();
53          return apu;
54      }
55  
56      @Override
57      public void piirra(Graphics g, int mittakaava, int x, int y, Hahmo pelaajanHahmo) {
58          int etaisyysHahmoonX = Math.abs(pelaajanHahmo.getSijainti().getX() - this.sijainti.getX());
59          int etaisyysHahmoonY = Math.abs(pelaajanHahmo.getSijainti().getY() - this.sijainti.getY());
60  
61          if (etaisyysHahmoonX > 1 || etaisyysHahmoonY > 1) {
62              g.setColor(Color.BLACK);
63          } else {
64              g.setColor(new Color(255, 255, 0));
65          }
66  
67          g.fillRect(x, y, mittakaava, mittakaava);
68      }
69  }
