160     /**
161      * Asettaa luolastoon annettujen koodinaattien kohdalle hirviön.
162      *
163      * Hirviö luodaan HirvionLuoja-luokkaa käyttäen metodille parametrina
164      * annetun koodin avulla.
165      *
166      * Jos asetus ei onnistu, palautetaan false. Muuten palautetaan true.
167      *
168      * @param x annettu x-koordinaatti
169      * @param y annettu y-koordinaatti
170      * @param hirvionLuontiKoodi annettu koodi, jolla hirviö luodaan
171      * @return onnistuiko hirviön asetus
172      */
173     public boolean asetaHirvio(int x, int y, int hirvionLuontiKoodi) {
174         if (x < 0 || x >= leveys || y < 0 || y >= korkeus) {
175             return false;
176         }
177 
178         Hirvio hh = HirvionLuoja.luoHirvio(hirvionLuontiKoodi);
179         hh.setSijainti(x, y);
180         koordinaatisto[x][y] = hh;
181 
182         return true;
183     }

387     /**
388      * Piirtää luolaston käyttöliittymän piirtotaululle. Metodi kutsuu
389      * luolatossa olevien kohteiden piirtometodeja. Kohteita piirretään sen
390      * mukaan, ovatko ne tarpeeksi lähellä pelaajan hahmoa. Jos kohde on liian
391      * kaukana, ei sitä piirretä. Tämän vuoksi pelaajan hahmoa kuljetetaan
392      * piirtometodissa.
393      *
394      * @param g käyttöliittyymän grafiikkaolio
395      * @param mittaKaava mittakaava, jossa piirretään
396      * @param pelaajanHahmo pelaajan hahmo
397      */
398     public void piirra(Graphics g, int mittaKaava, Hahmo pelaajanHahmo) {
399         for (int i = 0; i < leveys; i++) {
400             for (int j = 0; j < korkeus; j++) {
401                 int x = i * mittaKaava;
402                 int y = j * mittaKaava;
403                 koordinaatisto[i][j].piirra(g, mittaKaava, x, y, pelaajanHahmo);
404             }
405         }
