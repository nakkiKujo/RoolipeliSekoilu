1   package juuri.sovelluslogiikka.maailma;

3   import java.awt.Color;
4   import java.awt.Graphics;
5   import juuri.sovelluslogiikka.hahmo.Hahmo;
6   
7   /**
8    * Seinä on Kohteen aliluokka. Seinä rajaa luolastoa ja antaa sille rakenteen.
9    * Pelaaja ei normaalisti voi kulkea seinän sisälle.
10   */
11  public class Seina extends Kohde {
12      
13      /**
14       * Luo uuden seinän.
15       */
16      public Seina() {
17          this.koodi = Kohde.SEINA;
18          this.nimi = "seinä";
19          this.tapahtuma = null;
20  
21          this.voikoKulkea = false;
22      }
23  
24      @Override
25      public void piirra(Graphics g, int mittakaava, int x, int y, Hahmo pelaajanHahmo) {
26          int etaisyysHahmoonX = Math.abs(pelaajanHahmo.getSijainti().getX() - this.sijainti.getX());
27          int etaisyysHahmoonY = Math.abs(pelaajanHahmo.getSijainti().getY() - this.sijainti.getY());
28  
29          if (etaisyysHahmoonX > 1 || etaisyysHahmoonY > 1) {
30              g.setColor(Color.BLACK);
31          } else {
32              g.setColor(new Color(153, 102, 0));
33          }
34  
35          g.fillRect(x, y, mittakaava, mittakaava);
36      }
37  }
