1   package juuri.sovelluslogiikka.hahmo;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   
6   /**
7    * Velho on yksi Hahmon perivistä aliluokista. Velholla on enemmän taikavoimaa
8    * peli alussa kuin muilla hahmoilla. Lisäksi velho saa peli alussa muutaman
9    * riimun taikojen loihtimiseen.
10   */
11  public class Velho extends Hahmo {
12  
13      /**
14       * Luo uuden Velho-hahmon. Parametrina annetaan hahmon nimi. Konstruktori
15       * kutsuu perittävän Hahmo-luokan kontruktoria.
16       *
17       * Lisäksi asetetaan myös velholle koodi.
18       *
19       * @param nimi hahmon nimi
20       */
21      public Velho(String nimi) {
22          super(nimi);
23          this.ammatti = Hahmo.VELHO;
24      }
25  
26      @Override
27      public void piirra(Graphics g, int mittaKaava) {
28          int x = this.sijainti.getX() * mittaKaava;
29          int y = this.sijainti.getY() * mittaKaava;
30  
31          g.setColor(Color.BLUE);
32  
33          g.fillOval(x, y, mittaKaava, mittaKaava);
34      }
35  }
