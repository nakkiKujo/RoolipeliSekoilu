695     /**
696      * Piirtää luolaston käyttöliittymän piirtotaululle. Metodi kutsuu
697      * luolatossa olevien kohteiden piirtometodeja. Kohteita piirretään sen
698      * mukaan, ovatko ne tarpeeksi lähellä pelaajan hahmoa. Jos kohde on liian
699      * kaukana, ei sitä piirretä. Tämän vuoksi pelaajan hahmoa kuljetetaan
700      * piirtometodissa. Lisäksi kutsuttavassa metodissa tarkistetaan, onko
701      * kyseinen kohde seinän tai oven takana. Tällöin kyseistä kohdetta ei
702      * piirretä.
703      *
704      * @param g käyttöliittyymän grafiikkaolio
705      * @param mittaKaava mittakaava, jossa piirretään
706      * @param pelaajanHahmo pelaajan hahmo
707      */
708     public void piirra(Graphics g, int mittaKaava, Hahmo pelaajanHahmo) {
709         for (int i = 0; i < leveys; i++) {
710             for (int j = 0; j < korkeus; j++) {
711                 int x = i * mittaKaava;
712                 int y = j * mittaKaava;
713                 koordinaatisto[i][j].piirra(g, mittaKaava, x, y, pelaajanHahmo, this);
714             }
715         }
716     }
