10  /**
11   * Tämä luokka toimii välikätenä käyttöliittymän ja sovelluslogiikan välillä.
12   * Tästä luokasta lähetetään käskyjä muun muassa luolaston ja hahmon luomiseen
13   * sekä hahmon liikuttamiseen. Se toimii eräänlaisena runkona koko ohjelman
14   * toiminnalle.
15   */
16  public class Peli {
17  
18      private Hahmo pelaajanHahmo;
19      private Luolasto luolasto;
20      private Hahmonliikuttaja liikuttaja;
21      private Hahmonluoja luoja;
22  
23      public Peli() {
24          this.liikuttaja = new Hahmonliikuttaja();
25          this.luoja = new Hahmonluoja();
26      }
27  
28      public Hahmo getPelaajanHahmo() {
29          return pelaajanHahmo;
30      }
31  
32      public Luolasto getLuolato() {
33          return luolasto;
34      }
35  
36      public Hahmonliikuttaja getLiikuttaja() {
37          return liikuttaja;
38      }
39  
40      /**
41       * Metodi piirtää pelin käyttöliittymän piirtoalustalle. Se lähettää kutsun
42       * luolaston piirtometodille sekä pelaajan hahmon piirtometodille.
43       *
44       * @param g käyttöliittymän piirtoalustan grafiikka
45       */
46      public void piirra(Graphics g) {
47          //TODO: laskea mittaKaava jokaisessa erillisessä tapauksessa
48          int mittaKaava = 23;
49          luolasto.piirra(g, mittaKaava);
50          pelaajanHahmo.piirra(g, mittaKaava);
51      }
52  
53      /**
54       * Metodi luo pelaajan hahmon ja pelin maailman eli luolaston. Hahmo luodaan
55       * annetun nimen perusteella.
56       *
57       * @param hahmonNimi käyttäjän antama nimi hahmolle
58       */
59      public void aloitaPeli(String hahmonNimi) {
60          pelaajanHahmo = luoja.luoHahmo(hahmonNimi);
61  
62          this.luolasto = new Luolasto(9, 9);
63          luolasto.luoTaso1();
64      }
65  
66      public Tapahtuma liikutaanJaEtsitaanTapahtumaa(String suunta) {
67          //katsotaan, minkä kohteen hahmo kohtaa liikkuessaan
68          Kohde minneHahmoLiikkui = liikuttaja.liikutaHahmoa(suunta, luolasto, pelaajanHahmo);
69  
70          //Tapahtuma, joka alkaa hahmon kohdatessa kohteen. Jos tapahtuma on null, ei
71          //mitään tapahdu.
72          Tapahtuma tt = minneHahmoLiikkui.getTapahtuma();
73          if (tt == null) {
74              return null;
75          }
76  
77          return tt;
78      }
79  }
