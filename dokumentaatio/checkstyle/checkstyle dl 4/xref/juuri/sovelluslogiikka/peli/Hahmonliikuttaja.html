3   import juuri.apuvalineet.Sijainti;
4   import juuri.sovelluslogiikka.hahmo.Hahmo;
5   import juuri.sovelluslogiikka.maailma.Kohde;
6   import juuri.sovelluslogiikka.maailma.Luolasto;
7   
8   /**
9    * Tämän luokan vastuulla on liikuttaa pelaajan hahmoa.
10   */
11  public class Hahmonliikuttaja {
12  
13      public Hahmonliikuttaja() {
14      }
15  
16      /**
17       * Metodi palauttaa kohteen, johon hahmo liikkuessaan päätyy. Liikkuminen
18       * tapahtuu annetun komennon mukaisesti. Jos hahmo yrittää liikkua
19       * luolastosta ulos, palauttaa metodi arvon null.
20       *
21       * Jos hahmo yrittää liikkua kohteeseen, johon se ei voi mennä, kumotaan
22       * kyseinen toiminta. Metodi kuitenkin palauttaa kohteen, johon hahmo yritti
23       * kulkea.
24       *
25       * @param komento minne suuntaan hahmoa liikutetaan
26       * @param luola hahmo liikkuu luolastossa
27       * @param hahmo liikkuva hahmo
28       * @return kohde, johon hahmo liikkui
29       */
30      public Kohde liikutaHahmoa(String komento, Luolasto luola, Hahmo hahmo) {
31          Sijainti hahmonSijainti = hahmo.getSijainti();
32  
33          //ensin käsitellään komento, joka kertoo, mihin suuntaan liikutaan
34          if (komento.equals("alas")) {
35              hahmonSijainti.liikuAlas(1);
36          } else if (komento.equals("ylös")) {
37              hahmonSijainti.liikuYlos(1);
38          } else if (komento.equals("oikea")) {
39              hahmonSijainti.liikuOikealle(1);
40          } else if (komento.equals("vasen")) {
41              hahmonSijainti.liikuVasemmalle(1);
42          }
43  
44          Kohde loydetty = luola.haeSijainnista(hahmonSijainti);
45  
46          if (loydetty == null) {
47              return null;
48          }
49  
50          if (!loydetty.getVoikoKulkea()) {
51              if (komento.equals("alas")) {
52                  hahmonSijainti.liikuAlas(-1);
53              } else if (komento.equals("ylös")) {
54                  hahmonSijainti.liikuYlos(-1);
55              } else if (komento.equals("oikea")) {
56                  hahmonSijainti.liikuOikealle(-1);
57              } else if (komento.equals("vasen")) {
58                  hahmonSijainti.liikuVasemmalle(-1);
59              }
60          }
61  
62          return loydetty;
63      }
64  }
