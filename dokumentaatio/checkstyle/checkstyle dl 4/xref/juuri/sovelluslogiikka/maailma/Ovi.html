3   import juuri.sovelluslogiikka.tapahtumat.Tapahtuma;
4   
5   /**
6    * Ovi on Kohteen aliluokka. Ovi voi olla avoin, jolloin pelaaja voi normaalisti
7    * kulkea sen läpi. Ovi voi myös olla lukossa, jolloin sen läpi ei voi kulkea
8    * ilman esimerkiksi avainta.
9    */
10  public class Ovi extends Kohde {
11  
12      private boolean lukittu;
13  
14      public Ovi(boolean lukitus, Tapahtuma tapahtuma) {
15          this.koodi = Kohde.OVI;
16          this.nimi = "ovi";
17          this.lukittu = lukitus;
18  
19          //Ovien läpi kulkeminen riippuu sen lukituksesta.
20          if (lukitus) {
21              this.voikoKulkea = false;
22          } else {
23              this.voikoKulkea = true;
24          }
25          this.tapahtuma = tapahtuma;
26      }
27  
28      public boolean onkoLukittu() {
29          return lukittu;
30      }
31  
32  }
