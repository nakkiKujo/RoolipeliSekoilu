1   package juuri.sovelluslogiikka.hahmo;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   import juuri.apuvalineet.Sijainti;
6   
7   /**
8    * Tämä abstrakti luokka kuvastaa pelaajan pelissä seikkailevaa hahmoa. Luokalla
9    * on aliluokkina ammatit, jotka määräävät, millainen hahmo on. Esimerkiksi
10   * soturi on hahmona erilainen verrattuna velhoon. TODO: Ammattien käsittely,
11   * level-up.
12   */
13  public abstract class Hahmo {
14  
15      protected String nimi;
16      protected String ammatti;
17      protected Sijainti sijainti;
18      protected int voima;
19      protected int taikaVoima;
20      protected int elamaPisteet;
21      protected int ketteryys;
22  
23      /**
24       * Hahmon sijainti on pelin alussa aina (1,1). Hahmo luodaan pelaajan
25       * syöttämän nimen mukaisesti.
26       *
27       * @param nimi Käyttäjän antama nimi hahmolle
28       */
29      public Hahmo(String nimi) {
30          this.nimi = nimi;
31          this.sijainti = new Sijainti();
32          sijainti.setSijainti(1, 1);
33      }
34  
35      public String getNimi() {
36          return this.nimi;
37      }
38  
39      public String getAmmatti() {
40          return this.ammatti;
41      }
42  
43      public Sijainti getSijainti() {
44          return this.sijainti;
45      }
46  
47      /**
48       * Piirtää hahmon käyttöliittymän piirtoalustalle.
49       *
50       * @param g Piirtoalustaan liittyvä grafiikka.
51       * @param mittaKaava Määrää, minkä kokoisena hahmo piirretään.
52       */
53      public void piirra(Graphics g, int mittaKaava) {
54          int x = this.sijainti.getX() * mittaKaava;
55          int y = this.sijainti.getY() * mittaKaava;
56  
57          g.setColor(Color.red);
58  
59          g.fillOval(x, y, mittaKaava, mittaKaava);
60      }
61  }
